$.tinyscroller - callbacks

$(function(){

    // set callbacks like below
    $('a[href^=#]').tinyscrollable({
        scrollstart: function(){
            $.fixedConsole.log('start handler executed.');
        },
        scrollend: function(){
            $.fixedConsole.log('end handler executed.');
        }
    });

    /*
        can set globally like below.
        "everyscrollstart" and "scrollstart" are same.
        "everyscrollend" and "scrollend" are same.
        use "every-" callback for stable callbacks.
    */
    $.tinyscroller.setOptions({
        everyscrollstart: function(){
            $.fixedConsole.log('everyscrollstart handler executed.');
        },
        everyscrollend: function(){
            $.fixedConsole.log('everyscrollend handler executed.');
        }
    });
