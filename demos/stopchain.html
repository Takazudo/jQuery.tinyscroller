Stop

stop returns deferred object which watches stopping scrolling was done.

var scroller = $.Tinyscroller();
$(function(){
  $('#btn1').click(function(){
    scroller.scrollTo(3000).done(function(){
      alert('done!'); // but will not work
    });
    setTimeout(function(){
      scroller.stop().done(function(){
        scroller.scrollTo(0).done(function(){
          alert('stop and did 2nd scroll!');
        });
      });
    }, 200);
  });
});
